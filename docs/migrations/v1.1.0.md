## TASK 1 — Write scripts/migrate-records.js

Create this file. It must:

1. Read every AVE-*.json in records/
2. For each record, apply ALL of the following transforms in order:
   a. schema_version: "0.2.0" → "1.0.0"
   b. severity: if missing at top level, copy from r.aivss.aivss_severity
   c. aivss_score: if missing at top level, copy from r.aivss.aivss_score
   d. status: if missing, set "active"
   e. published: if missing, set "2026-04-14T09:00:00Z"
   f. researcher: if missing, set "Bawbel Security Research Team"
   g. researcher_url: if missing, set "https://bawbel.io"
   h. references: for each item that is a plain string, convert to
      { "tag": "Reference", "text": <the string>, "url": <the string> }
   i. owasp_mcp: if missing at top level, copy from r.aivss.owasp_mcp_mapping
   j. component_type: normalise old 0.2.0 values using this map:
      "mcp"             → "mcp_server"
      "mcp-server-card" → "mcp_server"
      "system_prompt"   → "prompt"
      "rag"             → "other"  (RAG pipeline is not a component type)
      "skill"           → "skill"  (already valid — keep as-is)
      "prompt"          → "prompt" (already valid — keep as-is)
      any other valid enum value   → keep as-is
      Valid v1.0.0 values: skill, prompt, mcp_server, plugin, agent, tool, other
3. Write the updated JSON back to the same file (in-place, 2-space indent)
4. Validate each written record against schema/ave-record-1.0.0.schema.json
5. Print: "ok   AVE-2026-NNNNN" or "FAIL AVE-2026-NNNNN: <message>"
6. Exit code 1 if any record fails validation

Use ajv/dist/2020 + ajv-formats (already installed).
Do NOT remove any fields that are in the schema's properties list.
Do NOT add fields that are not in the schema's properties list.

## TASK 5 — Write detection rules and fixtures for 5 priority records

### Step 5a — Create tests/test_fixtures.py if it does not exist

```python
import pytest
from pathlib import Path
import importlib.util

FIXTURES = Path("tests/fixtures")
RULES    = Path("rules/pattern")

def load_rule(ave_id):
    for f in RULES.glob("*.py"):
        spec = importlib.util.spec_from_file_location(f.stem, f)
        mod  = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(mod)
        if mod.RULE.get("ave_id") == ave_id:
            return mod
    return None

def fixture_pairs():
    for pos in sorted(FIXTURES.glob("*_positive.*")):
        ave_id = pos.stem.replace("_positive", "")
        neg = FIXTURES / pos.name.replace("_positive", "_negative")
        if neg.exists():
            yield ave_id, pos, neg

@pytest.mark.parametrize("ave_id,pos,neg", list(fixture_pairs()))
def test_positive(ave_id, pos, neg):
    mod = load_rule(ave_id)
    if mod is None:
        pytest.skip(f"no rule for {ave_id}")
    assert mod.matches(pos.read_text()), \
        f"{ave_id}: positive fixture did not trigger"

@pytest.mark.parametrize("ave_id,pos,neg", list(fixture_pairs()))
def test_negative(ave_id, pos, neg):
    mod = load_rule(ave_id)
    if mod is None:
        pytest.skip(f"no rule for {ave_id}")
    assert not mod.matches(neg.read_text()), \
        f"{ave_id}: negative fixture triggered (false positive)"
```

### Step 5b — For each priority record, write:

  rules/pattern/AVE-2026-NNNNN.py
  tests/fixtures/AVE-2026-NNNNN_positive.md   ← MUST trigger
  tests/fixtures/AVE-2026-NNNNN_negative.md   ← MUST NOT trigger

Rule file structure:

```python
import re

# What: pattern rule for <attack class>
# Why:  detects behavioral IOCs defined in AVE-2026-NNNNN
# How:  regex patterns matched against skill file text content

RULE = {
    "rule_id": "bawbel-<kebab-attack-class>",
    "ave_id":  "AVE-2026-NNNNN",
    "patterns": [
        re.compile(r"<pattern derived from indicators_of_compromise>", re.I),
    ],
}

def matches(content: str) -> list[str]:
    return [p.pattern for p in RULE["patterns"] if p.search(content)]
```

Derive patterns from the indicators_of_compromise field of each record.
The negative fixture is the most important — it must be a realistic benign
skill file that looks similar to the malicious one but lacks the IOC.

### Coverage check script

Write this as scripts/check_rule_coverage.py:

```python
# What: checks every AVE record in records/ has a rule in rules/pattern/
# Why:  a record with no rule is a definition nobody can detect
# How:  compares AVE ids in records/ against ave_id values in rules/pattern/

import os, re, sys

records = sorted(f.replace(".json","") for f in os.listdir("records") if f.endswith(".json"))
covered = set()

for f in os.listdir("rules/pattern"):
    if not f.endswith(".py") or f.startswith("_"):
        continue
    content = open(f"rules/pattern/{f}").read()
    m = re.search(r'"ave_id"\s*:\s*"(AVE-\d{4}-\d{5})"', content)
    if m:
        covered.add(m.group(1))

missing = [r for r in records if r not in covered]
if missing:
    print(f"Missing rules for {len(missing)} records:")
    for m in missing:
        print(f"  {m}")
    sys.exit(1)
else:
    print(f"All {len(records)} records have detection rules.")
```

Run: `python scripts/check_rule_coverage.py`

### Fixture coverage check

Write this as scripts/check_fixtures.py:

```python
# What: checks every rule in rules/pattern/ has positive and negative fixtures
# Why:  a rule without a negative fixture has no false-positive guard
# How:  checks tests/fixtures/ for AVE-YYYY-NNNNN_positive.* and _negative.*

import os, re, sys

rules_dir   = "rules/pattern"
fixture_dir = "tests/fixtures"

rule_ids = []
for f in os.listdir(rules_dir):
    if not f.endswith(".py") or f.startswith("_"):
        continue
    content = open(f"{rules_dir}/{f}").read()
    m = re.search(r'"ave_id"\s*:\s*"(AVE-\d{4}-\d{5})"', content)
    if m:
        rule_ids.append(m.group(1))

errors = []
for ave_id in sorted(rule_ids):
    pos = any(f.startswith(f"{ave_id}_positive") for f in os.listdir(fixture_dir))
    neg = any(f.startswith(f"{ave_id}_negative") for f in os.listdir(fixture_dir))
    if not pos:
        errors.append(f"{ave_id}: missing positive fixture")
    if not neg:
        errors.append(f"{ave_id}: missing negative fixture")

if errors:
    for e in errors: print(e)
    sys.exit(1)
else:
    print(f"All {len(rule_ids)} rules have positive and negative fixtures.")
```

## TASK 10 — AVE-in-SARIF convention

File: docs/specs/ave-in-sarif.md

This document specifies how AVE findings travel as SARIF so they reach the
GitHub Security tab and CI systems without any extra integration work.
A scanner that emits correct AVE-in-SARIF gets GitHub Security tab integration
for free.

Write the document covering:

### Required SARIF fields for AVE findings

```
run.tool.driver.rules[].id              = ave_id (e.g. "AVE-2026-00001")
run.tool.driver.rules[].name            = attack_class
run.tool.driver.rules[].shortDescription.text = title
run.tool.driver.rules[].fullDescription.text  = description
run.tool.driver.rules[].helpUri         = "https://ave.bawbel.io/registry.html#<ave_id>"
run.tool.driver.rules[].properties.severity   = severity
run.tool.driver.rules[].properties.aivss_score = aivss.aivss_score

result.ruleId                           = ave_id
result.message.text                     = behavioral_fingerprint + match location
result.level                            = "error" (CRITICAL/HIGH) | "warning" (MEDIUM) | "note" (LOW)
result.locations[].physicalLocation     = file path + line number
result.properties.confidence            = <per-detection float, NOT from record>
result.properties.evidence_kind         = <per-detection, from evidence_kind_default>
result.properties.owasp_mcp             = owasp_mcp array
result.properties.mitre_atlas           = mitre_atlas_mapping array
```

### SARIF taxonomies block

```json
"taxonomies": [{
  "name": "AVE",
  "version": "1.0.0",
  "informationUri": "https://ave.bawbel.io",
  "taxa": [
    { "id": "AVE-2026-00001", "name": "Supply Chain - Metamorphic Payload" }
  ]
}]
```

### Severity mapping

| AVE severity | SARIF level |
|---|---|
| CRITICAL | error |
| HIGH | error |
| MEDIUM | warning |
| LOW | note |

### Key rule: confidence is per-finding, never per-rule

SARIF `run.tool.driver.rules` carries the static record data.
SARIF `result.properties.confidence` carries the per-detection confidence.
Never put confidence in the rules block.
